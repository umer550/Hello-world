package assignment;
import java.io.*;
/**
 *
 * @author hp
 */
public class Assignment {

    /**
      @param args the command line arguments
     */
    int num;
    String first="abc";
    String second="foo";
    
    void printing_func(int i){
    if(i%3==0)
        System.out.println(first);
    else if(i%5==0)
         System.out.println(second);
    else
          System.out.println(i);}
    public static void main(String[] args) {
        // TODO code application logic here
        int rem,sum;
        sum=0;
        Assignment obj1=new Assignment();
        obj1.num=200;
        while(obj1.num<200){
        for(int i=1;i<=200;i++){
           rem=i%10;
           if(rem==3||rem==5)
               obj1.printing_func(i);
           else{ 
               while(rem!=0) {
                   sum+=rem;
               rem/=10;
               rem%=10;
               }
               obj1.printing_func(sum);
        }
        }
        }
    }
    
}
